Table employees {
  EmployeeID integer [primary key]
  FirstName varchar
  LastName varchar
  Email varchar
  Role varchar
  BirthDate date
  HireDate date
  Address varchar
  City varchar
  PostalCode varchar
  Country varchar
  Phone varchar
}

Table students {
  StudentID integer [primary key]
  FirstName varchar
  LastName varchar
  Email varchar
  Password varchar
}

Table lecturers {
  LecturerID integer [primary key]
  FirstName varchar
  LastName varchar
  Email varchar
  BirthDate date
  HireDate date
  Address varchar
  City varchar
  PostalCode varchar
  Country varchar
  Phone varchar
}

Table modules {
  ModuleID integer [primary key]
  EventID integer 
  ModuleNo integer 
  Description varchar
  BeginDate date
  EndDate date
  RoomID integer
  ModulePrice float 
}

Table events {
  EventID integer [primary key]
  Title varchar
  Type varchar
  BeginDate date
  EndDate date
  Capacity integer
  EventPrice float 
  Description varchar
}

Table rooms {
  RoomID integer [primary key]
  Name varchar
  Capacity integer
  Availability boolean
  Equipment varchar
}

Table event_lecturers {
  EventID integer [primary key]
  LecturerID integer [primary key]
}

Table event_employees {
  EventID integer [primary key]
  EmployeeID integer [primary key]
}

Table module_students {
  ModuleStudentID integer [primary key]
  ModuleID integer
  StudentID integer
  LeftToPay float
  isPresent boolean //null oznacza, że moduł się jeszcze nie odbył lub brak informacji, true to obecny, false to nieobecny 
}

Table employee_regions {
  RegionID integer [primary key]
  EmployeeID integer
}

Table lecturer_regions {
  RegionID integer [primary key]
  LecturerID integer 
}

Table region {
  RegionID integer [primary key]
  Region varchar
  RegionDescription varchar
}

Ref: event_employees.EmployeeID < employees.EmployeeID

Ref: students.StudentID < module_students.StudentID
Ref: module_students.ModuleID > modules.ModuleID

Ref: employees.EmployeeID < employee_regions.EmployeeID
Ref: lecturers.LecturerID < lecturer_regions.LecturerID

Ref: region.RegionID - employee_regions.RegionID
Ref: region.RegionID - lecturer_regions.RegionID
